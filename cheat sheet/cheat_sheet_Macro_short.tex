\subsection*{AD $\Leftrightarrow$ SM}
The interest rate and price stream are still inter-determined:
$$\frac{1}{1+\tilde{r}_{t+1}}=\frac{\hat{p}_{t+1}}{\hat{p}_t},\ \frac{1}{1+\tilde{r}_1}=\hat{p}_1$$

The two Euler equations are:
\begin{align*}
    u'_t(c_t^t)&=\beta u'_{t+1}(c_{t+1}^t)(\hat{p}_t/\hat{p}_{t+1}) &\cdots \text{ AD}\\
    u'_t(c_t^t)&=\beta u'_{t+1}(c_{t+1}^t)(1+\tilde{r}_{t+1}) & \cdots\text{ SM}
\end{align*}

\underline{\textit{\textbf{An easy proof}}}:
\begin{itemize}
    \item[-] \textbf{\textit{AD $\Rightarrow$ SM}}: AD equilibrium allocation  satisfies SM FOC and the SM budget constraints: 
    $$\tilde{\mu}_t^t=\hat{\lambda}\hat{p}_t,\tilde{\mu}_{t+1}^t=\hat{\lambda}\hat{p}_{t+1},\forall t\geq 0;\  \tilde{s}^t_t=w^t_t-\hat{c}^t_t,\forall t\geq 1$$
    \item[-] \textbf{\textit{SM $\Rightarrow$ AD}}: SM equilibrium allocation satisfies AD FOC and the AD budget constraints:  $$\hat{\lambda}^t=\frac{\tilde{\mu}^t_t}{\prod^{t-1}_{\tau=0}(1+\tilde{r}_{t-\tau})}=\frac{\tilde{\mu}^t_{t+1}}{\prod^{t-1}_{\tau=0}(1+\tilde{r}_{t+1-\tau})},\forall t\geq 1$$
    
    $$\hat{\lambda}^0=\tilde{\mu}^0_1(1+\tilde{r}_1)$$
\end{itemize}

    $$\sum^{\infty}_{t=1}\prod_{\tau}^t\left(1+r_{\tau+1}\right) =+\infty$$
    
    This \underline{\textbf{includes}} two scenarios (PE):
    \begin{itemize}
        \item[-] \textbf{AD prices falling}, i.e., \textbf{positive interest rate} $r_{t+1}$
        \begin{itemize}
            \item[-] \underline{classic Autarky}
        \end{itemize}
        
        \item[-] \textbf{AD prices constant}, i.e., \textbf{zero interest rate} $r_{t+1}$
        \begin{itemize}
            \item[-] \underline{knife-edge Autarky}, \underline{all knife-edge MNSE} (converge to Autarky) 
            \item[-]\underline{classic MSE}, \underline{all classic MNSE} (converge to classic MSE)
            \item[-] \underline{Samuelson MSE}
        \end{itemize}
    \end{itemize}
    This \underline{\textbf{excludes}} one scenario (not PE):
    \begin{itemize}
        \item[-] \textbf{AD prices increasing}, i.e., \textbf{negative interest rate} $r_{t+1}$ 
        \begin{itemize}
            \item[-] \underline{Samuelson Autarky}, \underline{all Samuelson MNSE} (converge to Samuelson Autarky). 
        \end{itemize}
    \end{itemize}
    \end{minipage}

\subsubsection*{competitive equilibrium}
Use \textbf{SM equilibrium}. The HH's Bellman equation is
$$

$$
s.t.
$$

$$
and the recursive competitive equilibrium is
\begin{itemize}
    \item[-] value function: $v(k,K)$
    \item[-] policy function: $c=C(k,K)$, $k'=G(k,K)$
    \item[-] pricing function: $w(K),r(K)$
    \item[-] aggregate law of motion: $K'=H(K)$
\end{itemize}
s.t.
\begin{itemize}
    \item[-] given pricing functions, \textbf{value function solves the Bellman equation}, with $C,G$ the associated policy functions
    \item[-] \textbf{pricing function maximize firm's profit}
    \item[-] \textbf{consistency}: $H(K)=K'=G(K,K)$
    \item[-] \textbf{market clearing}: $C(K,K)+G(K,K)-(1-\delta)K=F(K,1)$
\end{itemize}

\section*{Balanced growth}
\textbf{\color{myred}\underline{3 conditions} }to ensure balanced growth:
\begin{itemize}
    \item[-] \textbf{strictly positive growth}: $\gamma_c >0$
    \item[-] \textbf{life-time utility bounded}: $\sum^{\infty}_{t=0}\beta^tu(c_t) = \sum^{\infty}_{t=0}\beta^t u(\gamma^tc_0)>\infty$
    \item[-] \textbf{transversality}: $\lim_{t\rightarrow \infty} \beta^t u'(c_t)k_t =0$
\end{itemize}

\subsection*{Romer externality}
\begin{itemize}

\item[-] consumers: $$

\subsection*{Asset pricing}
Household's problem is
$$
\max_{c(s^t),a(s^{t+1})} \sum_{t,s^t}\beta^t \Pr(s^t)u(c(s^t))
$$
s.t.
\begin{align*}
    c(s^t)+q(s^t)\cdot a_{t+1}(s^t) &\leq W(s^t)\\
    W(s^{t+1}) &= y(s^{t+1})+\left( q(s^{s+1}) + d(s^{t+1}) \right)\cdot a_{t+1}(s^t)
\end{align*}
complete market and AD security ($d=1,q=0$):
$$
c(s^t)+q(s^t)\cdot a_{t+1}(s^t) \leq y(s^t)+a_t(s^{t-1})
$$
get Euler
$$
u'(c(s^t)) = \beta \sum_{s_{t+1}}\Pr(s_{t+1}\mid s^t)u'(c(s^t,s_{t+1}))\frac{q(s^{t+1})+d(s^{t+1})}{q(s^t,s_{t+1})}
$$

\node [bluebox] (box){%

$$
1 = \beta \mathbb{E}_t\left[\frac{u'(c_{t+1})}{u'(c_t)} R_{t+1} \right]
$$
This is true for all asset, hence:
\begin{itemize}
        \item[-] Pricing kernel: $m_{t+1} = \beta \frac{u'(c(s^t,s_{t+1}))}{u'(c(s^t))}$
        \item[-] risk-free rate: $R^f_{t+1}=\frac{1}{\mathbb{E}_t\left[m_{t+1}\right]}$
        \item[-] pricing formula:  $\mathbb{E}_t(R_{t+1}) = R^f_{t+1} - R^f_{t+1}\mathrm{Cov}(m_{t+1},R_{t+1})$
        \item[-] risk-neutral probabilities: $$Q(s_{t+1}\mid s^t)=\frac{\Pr(s_{t+1}\mid s^t)u'\left(c'(s_{t+1},s^t)\right)}{\sum_{s_{t+1}}\Pr(s_{t+1}\mid s^t)u'\left(c'(s_{t+1},s^t)\right)}$$
        \item[-] Sharpe ratio: $s^j = \frac{\mathbb{E}(R^j_{t+1})-R^f_{t+1}}{\sigma(R^j_{t+1})}$ and $\lvert s_j \rvert\leq \frac{\sigma(m_{t+1})}{\mathbb{E}(m_{t+1})}$
    \end{itemize}

\end{itemize}

    \begin{itemize}
        \item[-] Euler equation: $C_t^{-\gamma} = \beta\mathbb{E}\left[C_{t+1}^{-\gamma}\frac{1+i_{t+1}}{1+\pi_{t+1}}\right]$
        \item[-] labor condition: $\psi L_t^{\phi} = \frac{W_t}{P_t}C_t^{-\gamma}$
        \item[-] market clearing: $Y_t= C_t$
        \item[-] GDP with misallocation: $Y_t=\frac{A_tL_t}{\Delta_t}$ where the productivity/misallocation $\Delta_t = (1-\mu)\left(\frac{1-\mu(1+\pi_t)^{\epsilon-1}}{1-\mu}\right)^{\frac{\epsilon}{\epsilon-1}}+(1+\pi_t)^{\epsilon}\mu\Delta_{t-1}$
        \item[-] firm pricing: $\frac{P_{j,t}}{P_t}=\frac{\epsilon}{\epsilon-1}\frac{X_t}{Z_t}$ where $X_t\equiv \chi_t Y_t^{1-\gamma}+\beta\mu\mathbb{E}_t\left[X_{t+1}\left(\frac{P_{t+1}}{P_t}\right)^{\epsilon}\right]$, $Z_t\equiv Y_t^{1-\gamma}+\beta\mu\mathbb{E}_t\left[Z_{t+1}\left(\frac{P_{t+1}}{P_t}\right)^{\epsilon-1}\right]$
        \item[-] real marginal cost: $\chi_t = \frac{W_t}{P_tA_t}$
        \item[-] inflation: $\frac{1-\mu(1+\pi_t)^{\epsilon-1}}{1-\mu} = \left(\frac{\epsilon}{\epsilon-1}\frac{X_t}{Z_t}\right)^{1-\epsilon}$
    \end{itemize}

\section*{Incomplete market}
\subsection*{Aiyagari}
Household solves
$$
\max_{c(s^t),A(s^{t+1})}\sum^{\infty}_{t=0}\sum_{s^t}\beta^t \Pr(s^t)u(c(s^t))
$$
s.t.
\begin{align*}
    A(s^{t+1}) &= y(s_t)-c(s^t) +RA(s^t) \\
    A(s^t) &\geq -b & \text{borrowing bound}
\end{align*}
a natural borrowing constraint is: the lowest-income state $\underline{s}$ forever $b = -\frac{1}{1-R}y(\underline{s})$. 

Recursively, state variables are $A,s$, then
$$
V(A,s) = \max_{c,A'} u(c) + \beta\sum_{s'}\Pr(s'\mid s)V(A',s')
$$
s.t.
$$A'\leq y(s)-c+RA,\ A\geq -b$$
define \textbf{cash on hand} $x\equiv RA+y$, rewrite the problem with $x,s$ as state variables:
$$
V(x,s) = \max_{c,x'(s')} u(c) + \beta\sum_{s'}\Pr(s'\mid s)V(x'(s'),s')
$$
s.t.
$$x'(s')\leq R(x-c)+y(s'),\ c\leq x+b$$
FOCs are
$$
\begin{aligned}
u'(c)-\sum_{s}\lambda(s)R-\mu &= 0 \\
\beta \Pr(s'\mid s)\frac{\partial V(x'(s'),s')}{\partial x} - \lambda(s) &= 0
\end{aligned} 
$$
gives
$$
 u'(c) = \beta R\sum_{s'} \Pr(s'\mid s)\frac{\partial V(x'(s'),s')}{\partial x} + \mu
$$
and envelope theorem
$$
\frac{\partial V(x,s)}{\partial x} = u'(c(x,s))
$$
gives Euler equation
\begin{align*}
    u'(c) &= \beta R\sum_{s'}\Pr(s'\mid s)u'(c(x'(s'),s'))+\mu \\
    &{\color{myred}\geq \beta R\sum_{s'}\Pr(s'\mid s)u'(c(x'(s'),s'))}
\end{align*}


\subsection*{Insurance contract}
Planner chooses effort $\{e_t\}$ and consumption $\{c_t\}$ to \textbf{maximize profits s.t. to household's utility}

    $$

    $$

    $$

    $$

\begin{align*}

\end{align*}
plus envelope condition 
{\color{myblue}$
V'(w(s))=-\mu

then \textbf{continuation utility is the {\color{myblue}same} as current utility}.

\vspace{4pt}
\subsubsection*{Limited condition}
Better than autarky $v_{aut}=\sum^{\infty}_{t=0}\beta^t \left(\sum_s \Pr(s)u(c(s))\right) = \frac{\sum_s\Pr(s)u(c(s))}{1-\beta}$. The recursive problem is 
$$
V(w) = \max_{c(s),w'(s)}\sum_s \Pr(s) \left[ y(s)-c(s) + \beta V(w'(s)) \right]
$$
s.t.
\begin{align*}
    \sum_s \Pr(s)\left[ u(c(s))+\beta w'(s) \right] &= w \\
    u(c(s)) + \beta w'(s) &\geq u(y(s))+\beta v_{aut}, \forall s
\end{align*}
FOCs give
\begin{align*}
    u'(c(s)) &= -\frac{1}{V'(w'(s))} & V'(w'(s)) &= V'(w) -\frac{\lambda(s)}{\Pr(s)}
\end{align*}

\subsubsection*{Two-sided limited commitment}
Planner needs a positive profit, the recursive problem is 
$$
V(w) = \max_{c(s),w'(s)}\sum_s \Pr(s) \left[ y(s)-c(s) + \beta V(w'(s)) \right]
$$
s.t.
\begin{align*}
    \sum_s \Pr(s)\left[ u(c(s))+\beta w'(s) \right] &= w \\
    u(c(s)) + \beta w'(s) &\geq u(y(s))+\beta v_{aut}, \forall s \\
    y(s)-c(s) +\beta V(w'(s)) & \geq 0, \forall s
\end{align*}
FOCs give
\begin{align*}
    u'(c(s)) &= -\frac{1}{V'(w'(s))} & V'(w'(s)) &= \frac{V'(w)\Pr(s)-\lambda(s)}{\Pr(s)+\eta(s)}
\end{align*}

\subsubsection*{moral hazard}
Both parties commit, but household exerts unobserved effort, the recursive problem is
$$
V(w) = \max_{e, c(s),w'(s)}\sum_s \Pr(s\mid e) \left[ y(s)-c(s) + \beta V(w'(s)) \right]
$$
s.t.
\begin{align*}
    &\sum_s \Pr(s\mid e)\left[ u(c(s))-e+\beta w'(s) \right] = w \\
    &\sum_s \Pr(s\mid e)\left[ u(c(s))-e+\beta w'(s) \right] \\
    &\geq \sum_s \Pr(s\mid \tilde{e})\left[ u(c(s))-\tilde{e}+\beta w'(s) \right], \forall \tilde{e}
\end{align*}
assume the special case: $e\in\left\{0,1\right\}$, FOCs give
\begin{align*}
    u'(c(s)) &= -\frac{1}{V'(w'(s))} & V'(w'(s)) &= V'(w) - \lambda\left[ 1-\frac{\Pr(s\mid 0)}{\Pr(s\mid 1)} \right]
\end{align*}
which gives the inverse Euler equation:
$$
\mathbb{E}\left[\frac{1}{u'(c_{t+1})}\right] = \frac{1}{u'(c_t)}
$$
$c\rightarrow 0$: agents will quit.

\subsubsection*{hidden income}
Define transfer $\tau(s)\equiv c(s)-y(s)$, planner's problem (downward binding)
$$
V(w) = \max_{\tau(s),w'(s)}\sum_s \Pr(s) \left[ -\tau(s) + \beta V(w'(s)) \right]
$$
s.t.
\begin{align*}
    &\sum_s \Pr(s)\left[ u(\tau(s)+y(s))+\beta w'(s) \right] = w \\
    &u(\tau(s)+y(s)) + \beta w'(s) \\
    &\geq u(\tau(s-1)+y(s)) + \beta w'(s-1), \forall s>1
\end{align*}

FOC gives
$$
\Pr(s)\left[V'(w'(s))+\mu\right] = \lambda(s+1)-\lambda(s)
$$
sum over all s:
\begin{align*}
    \sum_s \Pr(s)\left[V'(w'(s))+\mu\right] &= \sum_s \left[\lambda(s+1)-\lambda(s)\right]\\
    \Rightarrow \sum_s\Pr(s)V'(w'(s)) &= -\mu = V'(w(s))\\
    \Rightarrow \mathbb{E}[V'(w'(s))] &= V'(w)
\end{align*}

\section*{Searching}
\begin{itemize}
    \item[-] \textbf{number of matches}: $h=m(u,v)$
    \item[-] \textbf{market tightness}: $\theta= \frac{v}{u}$
    \item[-] \textbf{vacancy filling rate}: $\frac{h}{v}=\frac{m(u,v)}{v}=m\left(\frac{1}{\theta},1\right)\equiv q(\theta)$
    \item[-] \textbf{job finding rate}: $\frac{h}{u} = \frac{h}{v}\frac{v}{u} = q(\theta)\theta = m(1,\theta)$, {\color{myred}$\partial\theta q(\theta)/\partial \theta>0$}
\end{itemize}
\subsection*{static version}
\begin{itemize}
    \item[-] bargaining: $
\max_{w} (w-b)^{\beta}(y-w)^{1-\beta} \Rightarrow w=\beta y+(1-\beta)b
$
\item[-] surplus
\begin{itemize}
    \item[-] worker: $w = b + \beta(y-b)$
    \item[-] firm: $-c+\frac{h}{v}(1-\beta)(y-b) = -c+q(\theta)(1-\beta)(y-b)=0$
\end{itemize}
\item[-] planner solves $
\max_v m(u,v)(y-b)-cv
$, FOC gives
$$
\frac{\partial m (u,v)}{\partial v} = \frac{c}{y-b} = q(\theta)(1-\beta) \Rightarrow 1-\beta = \frac{\partial m(u,v)}{\partial v}\frac{v}{m(u,v)}
$$
\end{itemize}

\subsection*{dynamic version}
\begin{itemize}
    \item[-] \textbf{Workers}' value
\begin{align*}
    rU &= b+\frac{h}{u}(E-U) +\dot{U}\\
    rE &= w+s(U-E) + \dot{E}
\end{align*}
\item[-] \textbf{Firms}' value
\begin{align*}
    rV &= -c+\frac{h}{v}(J-V)+\dot{V}\\
    rJ &= y-w + s(V-J) + \dot{J}
\end{align*}
\item[-] \textbf{surplus}
$$
S = E-U+J-V
$$
\item[-] bargaining result:
\begin{align*}
    E &= U + \beta S & \text{workers}\\
    J &= V + (1-\beta) S = (1-\beta)S & \text{firms} 
\end{align*}
\end{itemize}

Solve these, get
\begin{itemize}
    \item[-] $q(\theta)=\frac{c}{J}$: \textbf{more valuable} employees $J\uparrow$ leads to \textbf{tighter} market $\theta \uparrow$ 
    \item[-] $S=\frac{c}{(1-\beta)q(\theta)}$
    \item[-] $w=(1-\beta)b+\beta y+\beta\theta c$: \textbf{tighter} market $\theta\uparrow$ leads to \textbf{higher} wage $w\uparrow$
    \item[-] $\frac{r+s}{q(\theta)}+\beta\theta = \frac{(1-\beta)(y-b)}{c}$: \textbf{tightness} $\theta$ is \textbf{constant} over time.
    \item[-] the elasticity of $\theta$ w.r.t $y-b$, $\epsilon$ is smaller than empirical observations
\end{itemize}
